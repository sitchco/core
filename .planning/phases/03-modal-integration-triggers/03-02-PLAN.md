**3. Add modal Vimeo player creator (stores reference for pause/resume):**

**4. Add handleModalShow function:**

```javascript
function handleModalShow(modal) {
    var playerContainer = modal.querySelector('.sitchco-video__modal-player');
    if (!playerContainer) return; // Not a video modal

    var modalId = modal.id;
    var entry = modalPlayers.get(modalId);

    // Resume existing player
    if (entry && entry.player) {
        if (entry.provider === 'youtube') {
            entry.player.playVideo();
        } else {
            entry.player.play();
        }
        return;
    }

    // Prevent double-creation during SDK load
    if (entry && entry.loading) return;

    // First open: load SDK and create player
    var provider = playerContainer.dataset.provider;
    var videoId = playerContainer.dataset.videoId;
    var url = playerContainer.dataset.url;
    var startTime = provider === 'youtube'
        ? extractYouTubeStartTime(url)
        : extractVimeoStartTime(url);

    modalPlayers.set(modalId, { player: null, provider: provider, loading: true });

    if (provider === 'youtube') {
        createModalYouTubePlayer(playerContainer, videoId, startTime, modalId);
    } else if (provider === 'vimeo') {
        createModalVimeoPlayer(playerContainer, videoId, startTime, modalId);
    }
}
```

**5. Register ui-modal-show hook (priority 20, after UIModal core at 10):**

```javascript
sitchco.hooks.addAction('ui-modal-show', handleModalShow, 20, 'video-block');
```

This fires for ALL modal opens -- including deep link opens via syncModalWithHash() and trigger clicks via UIModal's delegated click handler. The function checks for `.sitchco-video__modal-player` to skip non-video modals.

**6. Register native close event listeners for pause:**

Inside the `sitchco.register()` block (alongside existing `initVideoBlocks`), add:

```javascript
// Pause video on modal close (all dismiss methods: Escape, backdrop, close button)
document.querySelectorAll('dialog.sitchco-modal--video').forEach(function (modal) {
    modal.addEventListener('close', function () {
        var entry = modalPlayers.get(modal.id);
        if (!entry || !entry.player) return;
        if (entry.provider === 'youtube') {
            entry.player.pauseVideo();
        } else {
            entry.player.pause();
        }
    });
});
```

CRITICAL: Use the native `close` event, NOT the `ui-modal-hide` hook. The hook does NOT fire on Escape key close (verified in UIModal source analysis). The native `close` event fires for all close methods.

**7. Update initVideoBlock to handle modal mode click:**

In `initVideoBlock(wrapper)`, replace the early return for non-inline modes with modal-mode handling:

```javascript
function initVideoBlock(wrapper) {
    var displayMode = wrapper.dataset.displayMode;

    if (displayMode === 'modal') {
        // Modal mode: click poster -> open modal (not inline play)
        var modalId = wrapper.dataset.modalId;
        var modal = document.getElementById(modalId);
        if (!modal) return;

        var clickBehavior = wrapper.dataset.clickBehavior;
        var clickTarget = clickBehavior === 'icon'
            ? wrapper.querySelector('.sitchco-video__play-button')
            : wrapper;
        if (!clickTarget) return;

        // Click handler (NOT { once: true } -- modal can be opened multiple times)
        clickTarget.addEventListener('click', function (e) {
            e.preventDefault();
            sitchco.hooks.doAction('ui-modal-show', modal);
        });

        // Keyboard handler for wrapper with role="button"
        if (wrapper.getAttribute('role') === 'button') {
            wrapper.addEventListener('keydown', function (e) {
                if (e.key === 'Enter' || e.key === ' ') {
                    e.preventDefault();
                    sitchco.hooks.doAction('ui-modal-show', modal);
                }
            });
        }
        return;
    }

    // Modal-only blocks have no on-page wrapper -- skip
    if (displayMode !== 'inline') {
        return;
    }

    // ... existing inline mode code (unchanged) ...
}
```

Key differences from inline mode:
- No `{ once: true }` on event listeners -- modal can be reopened multiple times
- Click calls `sitchco.hooks.doAction('ui-modal-show', modal)` instead of `handlePlay(wrapper)`
- No dimension locking, no `--playing` class -- poster stays visible (MODL-04)
